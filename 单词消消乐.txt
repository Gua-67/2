    // ---------- 语音朗读功能 (Web Speech API) ----------
    function speakText(text) {
        if (!text || text === "?" || text.trim() === "") return;
        // 如果浏览器不支持语音合成，静默失败
        if (!window.speechSynthesis) {
            console.warn("当前浏览器不支持语音合成");
            return;
        }
        // 根据文本内容选择语言：包含中文字符则用中文，否则用英文
        const containsChinese = /[\u4e00-\u9fa5]/.test(text);
        const lang = containsChinese ? "zh-CN" : "en-US";
        const utterance = new SpeechSynthesisUtterance(text);
        utterance.lang = lang;
        utterance.rate = 0.9;   // 语速稍慢，清晰
        utterance.pitch = 1.0;
        // 取消任何正在进行的语音避免重叠（可选，为了体验不强制取消）
        // window.speechSynthesis.cancel(); // 取消会导致正在读的停止，体验不好，不主动取消
        window.speechSynthesis.speak(utterance);
    }

    // ---------- 词库持久化 ----------
    function saveWordBankToLocal() {
        localStorage.setItem('wordMatchPairs', JSON.stringify(userWordPairs));
    }

    function loadWordBankFromLocal() {
        const stored = localStorage.getItem('wordMatchPairs');
        if (stored) {
            try {
                const parsed = JSON.parse(stored);
                if (Array.isArray(parsed) && parsed.length) {
                    userWordPairs = parsed;
                    return;
                }
            } catch(e) {}
        }
        userWordPairs = DEFAULT_WORD_PAIRS.map(p => ({ word: p.word, translation: p.translation }));
        saveWordBankToLocal();
    }

    // 刷新词库UI列表
    function renderWordList() {
        if (!wordListContainer) return;
        if (userWordPairs.length === 0) {
            wordListContainer.innerHTML = '<div style="text-align:center; padding:12px; color:#aa8b64;">暂无单词，点击「恢复默认」或导入Excel添加单词~</div>';
            wordCountBadge.innerText = `0 对`;
            return;
        }
        let html = '';
        userWordPairs.forEach((pair, idx) => {
            html += `
                <div class="word-item">
                    <span>📖 ${escapeHtml(pair.word)}  &nbsp;→&nbsp;  ${escapeHtml(pair.translation)}</span>
                    <button class="delete-word" data-index="${idx}">删除</button>
                </div>
            `;
        });
        wordListContainer.innerHTML = html;
        wordCountBadge.innerText = `${userWordPairs.length} 对`;
        // 绑定删除事件
        document.querySelectorAll('.delete-word').forEach(btn => {
            btn.addEventListener('click', (e) => {
                e.stopPropagation();
                const idx = parseInt(btn.getAttribute('data-index'), 10);
                if (!isNaN(idx) && idx >= 0 && idx < userWordPairs.length) {
                    userWordPairs.splice(idx, 1);
                    saveWordBankToLocal();
                    renderWordList();
                    messageBox.innerHTML = `📦 已删除词对，下次抽词生效。`;
                    setTimeout(() => {
                        if (!gameFinished && messageBox.innerHTML.includes("删除")) messageBox.innerHTML = "✨ 点击卡片，匹配单词 & 释义 ✨";
                    }, 1500);
                }
            });
        });
    }

    function escapeHtml(str) {
        return String(str).replace(/[&<>]/g, function(m) {
            if (m === '&') return '&amp;';
            if (m === '<') return '&lt;';
            if (m === '>') return '&gt;';
            return m;
        });
    }

    // 手动添加词对
    function addWordPair() {
        let word = wordInput.value.trim();
        let meaning = meaningInput.value.trim();
        if (!word || !meaning) {
            messageBox.innerHTML = "⚠️ 请同时填写英文和中文！";
            setTimeout(() => { if(messageBox && !gameFinished) messageBox.innerHTML = "✨ 点击卡片，匹配单词 & 释义 ✨"; }, 1000);
            return;
        }
        userWordPairs.push({ word, translation: meaning });
        saveWordBankToLocal();
        renderWordList();
        wordInput.value = '';
        meaningInput.value = '';
        messageBox.innerHTML = `✅ 已添加 "${word} ↔ ${meaning}"，点击「随机抽词」开始游戏！`;
        setTimeout(() => {

        }, 1500);
    }

    // ---------- Excel 解析与导入 ----------
    uploadExcelBtn.addEventListener('click', () => {
        excelUploadInput.click();
    });
    excelUploadInput.addEventListener('change', (event) => {
        const file = event.target.files[0];
        if (!file) {
            fileNameDisplay.innerText = "未选择文件";
            parsedExcelData = [];
            return;
        }
        fileNameDisplay.innerText = file.name;
        const reader = new FileReader();
        reader.onload = function(e) {
            const data = new Uint8Array(e.target.result);
            const workbook = XLSX.read(data, { type: 'array' });
            const firstSheet = workbook.Sheets[workbook.SheetNames[0]];
            const rows = XLSX.utils.sheet_to_json(firstSheet, { header: 1, defval: "" });
            if (!rows || rows.length === 0) {
                messageBox.innerHTML = "❌ Excel 文件无数据！";
                parsedExcelData = [];
                return;
            }
            let startRow = 0;
            if (rows.length > 0 && rows[0].length >= 2) {
                const firstRowCol1 = String(rows[0][0]).toLowerCase();
                const firstRowCol2 = String(rows[0][1]).toLowerCase();
                if (firstRowCol1.includes('word') || firstRowCol1.includes('英文') || firstRowCol1.includes('单词') ||
                    firstRowCol2.includes('meaning') || firstRowCol2.includes('中文') || firstRowCol2.includes('释义')) {
                    startRow = 1;
                }
            }
            const newPairs = [];
            for (let i = startRow; i < rows.length; i++) {
                const row = rows[i];
                if (!row || row.length < 2) continue;
                let word = row[0] ? String(row[0]).trim() : "";
                let translation = row[1] ? String(row[1]).trim() : "";
                if (word === "" || translation === "") continue;
                newPairs.push({ word, translation });
            }
            if (newPairs.length === 0) {
                messageBox.innerHTML = "⚠️ 未从Excel中解析出有效的单词对，请确保第一列单词，第二列释义。";
                parsedExcelData = [];
            } else {
                parsedExcelData = newPairs;
                messageBox.innerHTML = `📎 解析成功！共 ${parsedExcelData.length} 对单词，点击「导入并添加到词库」完成导入。`;
            }
        };
        reader.readAsArrayBuffer(file);
    });

    function importExcelToWordBank() {
        if (!parsedExcelData || parsedExcelData.length === 0) {
            messageBox.innerHTML = "❌ 请先选择并解析Excel文件！";
            return;
        }
        let addedCount = 0;
        for (const newPair of parsedExcelData) {
            const exists = userWordPairs.some(p => p.word === newPair.word && p.translation === newPair.translation);
            if (!exists) {
                userWordPairs.push({ word: newPair.word, translation: newPair.translation });
                addedCount++;
            }
        }
        if (addedCount > 0) {
            saveWordBankToLocal();
            renderWordList();
            messageBox.innerHTML = `✅ 成功导入 ${addedCount} 对新单词！当前词库共 ${userWordPairs.length} 对。`;
            parsedExcelData = [];
            excelUploadInput.value = '';
            fileNameDisplay.innerText = "未选择文件";
        } else {
            messageBox.innerHTML = `ℹ️ 没有新增单词（所有词对都已存在于词库中）。`;
        }
        setTimeout(() => {
            if (messageBox && !gameFinished && messageBox.innerHTML.includes("导入")) messageBox.innerHTML = "✨ 点击卡片，匹配单词 & 释义 ✨";
        }, 2000);
    }

    importExcelBtn.addEventListener('click', importExcelToWordBank);

    // ---------- 游戏核心 ----------
    function selectRandomPairsForGame() {
        if (userWordPairs.length < REQUIRED_PAIRS) {
            messageBox.innerHTML = `⚠️ 当前词库只有 ${userWordPairs.length} 对单词，至少需要 ${REQUIRED_PAIRS} 对才能开始游戏！请导入更多单词或恢复默认词库。`;
            return null;
        }
        const shuffledAll = [...userWordPairs];
        for (let i = shuffledAll.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [shuffledAll[i], shuffledAll[j]] = [shuffledAll[j], shuffledAll[i]];
        }
        const selected = shuffledAll.slice(0, REQUIRED_PAIRS);
        return selected.map(p => ({ word: p.word, translation: p.translation }));
    }

    function buildCardsFromPairs(pairs) {
        const newCards = [];
        let uid = 0;
        for (let i = 0; i < pairs.length; i++) {
            const pair = pairs[i];
            newCards.push({ id: uid++, pairId: i, text: pair.word, matched: false, isOpen: false });
            newCards.push({ id: uid++, pairId: i, text: pair.translation, matched: false, isOpen: false });
        }
        for (let i = newCards.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [newCards[i], newCards[j]] = [newCards[j], newCards[i]];
        }
        return newCards;
    }

    function resetGameByCurrentPairs() {
        if (!currentGamePairs.length) {
            const fallback = selectRandomPairsForGame();
            if (!fallback) return false;
            currentGamePairs = fallback;
        }
        if (timeoutId) clearTimeout(timeoutId);
        lockBoard = false;
        selectedIndexes = [];
        gameFinished = false;
        stepCount = 0;
        matchedCount = 0;
        remainingPairs = REQUIRED_PAIRS;
        updateStatsUI();
        stopTimer();
        timerSeconds = 0;
        isTimerRunning = false;
        timerDisplaySpan.innerText = "0";
        cards = buildCardsFromPairs(currentGamePairs);
        renderGrid();
        messageBox.innerHTML = "🎲 新对局！配对单词与释义吧！";
        setTimeout(() => {
            if (messageBox && !gameFinished) messageBox.innerHTML = "✨ 点击卡片，匹配单词 & 释义 ✨";
        }, 1200);
        return true;
    }

    function startNewGameWithRandomPairs() {
        const newPairs = selectRandomPairsForGame();
        if (!newPairs) return false;
        currentGamePairs = newPairs;
        return resetGameByCurrentPairs();
    }

    function updateStatsUI() {
        remainingSpan.innerText = remainingPairs;
        stepSpan.innerText = stepCount;
        timerDisplaySpan.innerText = timerSeconds;
    }

    function stopTimer() {
        if (timerInterval) { clearInterval(timerInterval); timerInterval = null; }
        isTimerRunning = false;
    }

    function startTimer() {
        if (timerInterval) stopTimer();
        isTimerRunning = true;
        timerInterval = setInterval(() => {
            if (isTimerRunning && !gameFinished) {
                timerSeconds++;
                timerDisplaySpan.innerText = timerSeconds;
            }
        }, 1000);
    }

    function checkVictory() {
        if (matchedCount === REQUIRED_PAIRS) {
            gameFinished = true;
            stopTimer();
            messageBox.innerHTML = "🏆 恭喜！全配对成功！点击「随机抽词」继续挑战新单词 🏆";
            messageBox.classList.add('win-glow');
            setTimeout(() => messageBox.classList.remove('win-glow'), 1000);
            lockBoard = true;
            return true;
        }
        return false;
    }

    function handleMatch(idxA, idxB) {
        const cardA = cards[idxA], cardB = cards[idxB];
        if (cardA.matched || cardB.matched) return false;
        cardA.matched = true; cardB.matched = true;
        cardA.isOpen = false; cardB.isOpen = false;
        matchedCount++;
        remainingPairs = REQUIRED_PAIRS - matchedCount;
        updateStatsUI();
        selectedIndexes = [];
        renderGrid();
        const isWin = checkVictory();
        if (!isWin) {
            messageBox.innerHTML = "✔️ 配对成功！继续～";
            setTimeout(() => {
                if (messageBox && !gameFinished && matchedCount !== REQUIRED_PAIRS) messageBox.innerHTML = "✨ 点击卡片，匹配单词 & 释义 ✨";
            }, 700);
        }
        return true;
    }

    function handleMismatch(idxA, idxB) {
        if (timeoutId) clearTimeout(timeoutId);
        lockBoard = true;
        timeoutId = setTimeout(() => {
            const cardA = cards[idxA], cardB = cards[idxB];
            if (cardA && !cardA.matched) cardA.isOpen = false;
            if (cardB && !cardB.matched) cardB.isOpen = false;
            selectedIndexes = [];
            renderGrid();
            lockBoard = false;
            timeoutId = null;
            if (!gameFinished && matchedCount !== REQUIRED_PAIRS) {
                messageBox.innerHTML = "🔄 不对应，再试试看～";
                setTimeout(() => {
                    if (messageBox && !gameFinished && matchedCount !== REQUIRED_PAIRS) messageBox.innerHTML = "✨ 点击卡片，匹配单词 & 释义 ✨";
                }, 800);
            }
        }, 700);
    }

    // 核心点击事件，增加读音功能
    function onCardClick(event) {
        let target = event.target.closest('.card');
        if (!target) return;
        const idx = parseInt(target.getAttribute('data-index'), 10);
        if (isNaN(idx)) return;
        const card = cards[idx];
        if (!card) return;

        // ---------- 读音功能：如果卡片是正面（翻开）或已配对，朗读文本 ----------
        // 注意：背面显示 "?" 不朗读，且锁板时暂不朗读（避免动画期间语音干扰）
        if (!lockBoard && (card.matched || card.isOpen)) {
            // 确保文本不为空且不是占位符
            if (card.text && card.text !== "?") {
                speakText(card.text);
            }
        }

        // 以下为原有游戏逻辑，部分情况会直接return，但读音已经执行
        if (gameFinished) { 
            messageBox.innerHTML = "🏆 游戏已胜利，请点击「随机抽词」开始新对局 🏆";
            return; 
        }
        if (lockBoard) return;
        if (card.matched) return;
        if (card.isOpen) return;

        if (!isTimerRunning && !gameFinished && matchedCount < REQUIRED_PAIRS) startTimer();

        if (selectedIndexes.length >= 2) return;
        card.isOpen = true;
        renderGrid();
        selectedIndexes.push(idx);
        if (selectedIndexes.length === 2) {
            stepCount++;
            updateStatsUI();
            const [i1, i2] = selectedIndexes;
            if (cards[i1].pairId === cards[i2].pairId) handleMatch(i1, i2);
            else handleMismatch(i1, i2);
        }
    }

    function renderGrid() {
        if (!gridContainer) return;
        gridContainer.innerHTML = '';
        for (let i = 0; i < cards.length; i++) {
            const c = cards[i];
            const div = document.createElement('div');
            div.classList.add('card');
            div.setAttribute('data-index', i);
            if (c.matched) {
                div.classList.add('matched');
                div.innerText = c.text;
                div.style.cursor = 'pointer';  // 已配对依然可点击听读音
            } else if (c.isOpen) {
                div.classList.add('front');
                div.innerText = c.text;
            } else {
                div.classList.add('back');
                div.innerText = '?';
            }
            gridContainer.appendChild(div);
        }
    }

    function reshuffleGame() {
        if (gameFinished) {
            if (currentGamePairs.length) resetGameByCurrentPairs();
            else startNewGameWithRandomPairs();
            return;
        }
        if (timeoutId) clearTimeout(timeoutId);
        lockBoard = false;
        selectedIndexes = [];
        if (currentGamePairs.length) {
            stopTimer();
            timerSeconds = 0; isTimerRunning = false;
            timerDisplaySpan.innerText = "0";
            stepCount = 0; matchedCount = 0; remainingPairs = REQUIRED_PAIRS;
            gameFinished = false;
            updateStatsUI();
            cards = buildCardsFromPairs(currentGamePairs);
            renderGrid();
            messageBox.innerHTML = "🃏 重新洗牌，继续挑战！";
            setTimeout(() => {
                if (messageBox && !gameFinished) messageBox.innerHTML = "✨ 点击卡片，匹配单词 & 释义 ✨";
            }, 1000);
        } else {
            startNewGameWithRandomPairs();
        }
    }

    // 事件绑定
    function bindEvents() {
        gridContainer.addEventListener('click', onCardClick);
        shuffleBtn.addEventListener('click', reshuffleGame);
        addWordBtn.addEventListener('click', addWordPair);
        resetDefaultBtn.addEventListener('click', resetToDefault);
        startGameBtn.addEventListener('click', () => {
            const success = startNewGameWithRandomPairs();
            if (!success && userWordPairs.length < REQUIRED_PAIRS) {
                messageBox.innerHTML = `⚠️ 词库不足 ${REQUIRED_PAIRS} 对，请导入Excel或添加单词。`;
            }
        });
        wordInput.addEventListener('keypress', (e) => { if(e.key === 'Enter') addWordPair(); });
        meaningInput.addEventListener('keypress', (e) => { if(e.key === 'Enter') addWordPair(); });
    }

    function init() {
        loadWordBankFromLocal();
        renderWordList();
        const initPairs = selectRandomPairsForGame();
        if (initPairs) currentGamePairs = initPairs;
        else if (userWordPairs.length >= REQUIRED_PAIRS) currentGamePairs = selectRandomPairsForGame();
        if (!currentGamePairs || currentGamePairs.length === 0) {
            userWordPairs = DEFAULT_WORD_PAIRS.map(p => ({ word: p.word, translation: p.translation }));
            saveWordBankToLocal();
            renderWordList();
            currentGamePairs = selectRandomPairsForGame();
        }
        resetGameByCurrentPairs();
        bindEvents();
    }

    init();
</script>
